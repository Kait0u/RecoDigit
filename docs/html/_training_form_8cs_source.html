218 private void LoadTrainingData(int trainingBatchSize)
219 {
220 {
221 List<int> labels = new List<int>();
222 List<List<double>> images = new List<List<double>>();
223
224 using (StreamReader reader = new StreamReader(datasetPath))
225 {
226 reader.ReadLine();
227
228 while (!reader.EndOfStream)
229 {
230 string line = reader.ReadLine();
231 string[] values = line.Split(",");
232
233 labels.Add(int.Parse(values[0]));
234
235 var tempImageString = values.Skip(1).ToList();
236 var tempImage = new List<double>();
237
238 foreach (var pixel in tempImageString)
239 {
240 tempImage.Add(int.Parse(pixel));
241 }
242
243 images.Add(tempImage);
244 }
245 }
246
247 double[][] imagesArray = new double[images.Count][];
248 for (int i = 0; i < images.Count; ++i)
249 {
250 imagesArray[i] = images[i].ToArray();
251 }
252
253 double[][] oneHotY = labels.Select(label => Utilities.OneHotEncode<double>(label, 10)).ToArray();
254 xTrain = imagesArray.Take(trainingBatchSize).Select(row => row.Select(pixel => pixel / 255.0).ToArray()).ToArray();
255 yTrain = oneHotY.Take(trainingBatchSize).ToArray();
256 }
257 }

263 private async void trainButton_Click(object sender, EventArgs e)
264 {
265 outputLogTextBox.Clear();
266 trainButton.Enabled = false;
267 trainingProgressBar.Value = 0;
268
269 Action<string?, int> trainingLogCallback = (string? message, int currIter) =>
270 {
271 if (message != null)
272 UpdateLogThread(message);
273
274 UpdateProgressBarThread(currIter);
275 };
276
277 try
278 {
279 GetData();
280 LoadTrainingData(samples);
281 trainingProgressBar.Maximum = iterations;
282 trainedNetwork = new NeuralNetwork(firstHiddenLayer, learningRate);
283 await Task.Run(() => trainedNetwork.Train(xTrain, yTrain, iterations, trainingLogCallback));
284 trainedNetwork.ExportToFile(outputModelPath);
285
286 }
287 catch (Exception ex)
288 {
289 UpdateLog(ex.Message);
290 trainButton.Enabled = true;
291 return;
292 }
293
294 UpdateLog("");
295 UpdateLog("Training complete! You may close this window now.");
296
297 bool load = autoLoad;
298
299 if (!load)
300 {
301 load = DialogResult.Yes == MessageBox.Show(
302 "Training complete! Would you like to load your network into the program?", "Training complete!", MessageBoxButtons.YesNo, MessageBoxIcon.Question);
303 }
304
305 if (load)
306 {
307 parent.Network = trainedNetwork;
308 parent.UpdateLog("Trained network loaded.", true);
309 Close();
310 }
311 }
