135 public double[] Predict(double[] x)
136 {
137 Z0 = new Matrix(x).Transpose();
138 Z1 = Matrix.DotProduct(W0, Z0) + B0;
139 A1 = ReLu(Z1);
140 Z2 = Matrix.DotProduct(W1, A1) + B1;
141 A2 = Sigmoid(Z2);
142
143 return A2.ToArray();
144 }

150 private void PredictMultiple(double[][] X)
151 {
152 Z0 = new Matrix(X).Transpose();
153 Z1 = Matrix.ColumnwiseBroadcastAdd(Matrix.DotProduct(W0, Z0), B0); // We need to add B0 to every column
154 A1 = ReLu(Z1);

165 private void BackPropagation(double[][] X, double[][] oneHotY)
166 {
167 int batchSize = X.Length;
168 PredictMultiple(X); // Make sure the values are fresh after making a prediction
169 Matrix targetY = new Matrix(oneHotY).Transpose();
170
171 Matrix dZ2 = A2 - targetY;
172 Matrix dW1 = Matrix.DotProduct(dZ2, A1.Transpose()) / batchSize;
173 double dB1 = dZ2.Sum() / batchSize;
174
175 Matrix dZ1 = Matrix.ElementProduct(Matrix.DotProduct(W1.Transpose(), dZ2), ReLuDer(Z1));
176 Matrix dW0 = Matrix.DotProduct(dZ1, Z0.Transpose()) / batchSize;
177 double dB0 = dZ1.Sum() / batchSize;
178
179 W1 = W1 - learningRate * dW1;
180 W0 = W0 - learningRate * dW0;
181 B1 = B1 - learningRate * dB1;
182 B0 = B0 - learningRate * dB0;
183 }

193 public void Train(double[][] X, double[][] Y, int iterations = 500, Action<string?, int> logCallback = null)
194 {
195 bool debug = logCallback != null;
196
197 int exampleCount = X.Length;
198 Initialize(exampleCount);
199
200 if (Y.Length != exampleCount) throw new InvalidOperationException("X and Y have to be of the same size for the training to proceed!");
201
202 for (int iterIdx = 0; iterIdx < iterations; ++iterIdx)
203 {
204 int successes = 0;
205 BackPropagation(X, Y);
206
207 for (int idx = 0; idx < exampleCount; ++idx)
208 {
209 int prediction = Utilities.ArgMax(A2.ExtractColumn(idx));
210 int actually = Utilities.ArgMax(Y[idx]);
211
212 if (prediction == actually) ++successes;
213 }
214
215 trainingSetAccuracy = (double)successes / exampleCount;
216 if (debug && iterIdx % 10 == 0) logCallback(string.Format("[ITERATION {0,6}]: ACCURACY = {1,6:F3}", iterIdx, trainingSetAccuracy), iterIdx + 1);
217 else if (debug) logCallback(null, iterIdx + 1);
218 }
219
220 if (debug) logCallback(string.Format("[FINAL RESULT]: ACCURACY = {0,6:F3}", trainingSetAccuracy), iterations);
221 }

228 public static Matrix ReLu(Matrix m)
229 {
230 int rowSize = m.Rows, colSize = m.Columns;
231 Matrix result = new Matrix(rowSize, colSize);
232
233 Parallel.For(0, rowSize, (r) =>
234 {
235 for (int c = 0; c < colSize; ++c)
236 {
237 result[r, c] = Math.Max(0, m[r, c]);
238 }
239 });
240
241 return result;
242 }

255 Parallel.For(0, rowSize, (r) =>
