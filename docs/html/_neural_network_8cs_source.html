133 public double[] Predict(double[] x)
134 {
135 Z0 = new Matrix(x).Transpose();
136 Z1 = Matrix.DotProduct(W0, Z0) + B0;
137 A1 = ReLu(Z1);
138 Z2 = Matrix.DotProduct(W1, A1) + B1;
139 A2 = Sigmoid(Z2);
140
141 return A2.ToArray();
142 }

163 private void BackPropagation(double[][] X, double[][] oneHotY)
164 {
165 int batchSize = X.Length;
166 PredictMultiple(X); // Make sure the values are fresh after making a prediction
167 Matrix targetY = new Matrix(oneHotY).Transpose();
168
169 Matrix dZ2 = A2 - targetY;
170 Matrix dW1 = Matrix.DotProduct(dZ2, A1.Transpose()) / batchSize;
171 double dB1 = dZ2.Sum() / batchSize;
172
173 Matrix dZ1 = Matrix.ElementProduct(Matrix.DotProduct(W1.Transpose(), dZ2), ReLuDer(Z1));
174 Matrix dW0 = Matrix.DotProduct(dZ1, Z0.Transpose()) / batchSize;
175 double dB0 = dZ1.Sum() / batchSize;
176
177 W1 = W1 - learningRate * dW1;
178 W0 = W0 - learningRate * dW0;
179 B1 = B1 - learningRate * dB1;
180 B0 = B0 - learningRate * dB0;
181 }

191 public void Train(double[][] X, double[][] Y, int iterations = 500, Action<string?, int> logCallback = null)
192 {
193 bool debug = logCallback != null;
194
195 int exampleCount = X.Length;
196 Initialize(exampleCount);
197
198 if (Y.Length != exampleCount) throw new InvalidOperationException("X and Y have to be of the same size for the training to proceed!");
199
200 for (int iterIdx = 0; iterIdx < iterations; ++iterIdx)
201 {
202 int successes = 0;
203 BackPropagation(X, Y);
204
205 for (int idx = 0; idx < exampleCount; ++idx)
206 {
207 int prediction = Utilities.ArgMax(A2.ExtractColumn(idx));
208 int actually = Utilities.ArgMax(Y[idx]);
209
210 if (prediction == actually) ++successes;
211 }
212
213 trainingSetAccuracy = (double)successes / exampleCount;
214 if (debug && iterIdx % 10 == 0) logCallback(string.Format("[ITERATION {0,6}]: ACCURACY = {1,6:F3}", iterIdx, trainingSetAccuracy), iterIdx + 1);
215 else if (debug) logCallback(null, iterIdx + 1);
216 }
217
218 if (debug) logCallback(string.Format("[FINAL RESULT]: ACCURACY = {0,6:F3}", trainingSetAccuracy), iterations);
219 }

226 public static Matrix ReLu(Matrix m)
227 {
228 int rowSize = m.Rows, colSize = m.Columns;
229 Matrix result = new Matrix(rowSize, colSize);
230
231 Parallel.For(0, rowSize, (r) =>
232 {
233 for (int c = 0; c < colSize; ++c)
234 {
235 result[r, c] = Math.Max(0, m[r, c]);
236 }
237 });
238
239 return result;
240 }

249 {
250 int rowSize = m.Rows, colSize = m.Columns;

252
253 Parallel.For(0, rowSize, (r) =>
254 {
255 for (int c = 0; c < colSize; ++c)
256 {

276 Parallel.For(0, rowSize, (r) =>
