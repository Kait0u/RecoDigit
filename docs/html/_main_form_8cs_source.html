208 private void recognizeButton_Click(object sender, EventArgs e)
209 {
210 if (neuralNetwork == null)
211 {
212 predictionTextBox.Text = "Load or train your neural network first!";
213 }
214 else
215 {
216 Bitmap resized = ImageUtilities.Resize(bitmap, outputImageSize, outputImageSize, System.Drawing.Drawing2D.InterpolationMode.NearestNeighbor);
217 resized = ImageUtilities.Resize(bitmap, outputImageSize * 2, outputImageSize * 2, System.Drawing.Drawing2D.InterpolationMode.Bicubic);
218 resized = ImageUtilities.Resize(bitmap, outputImageSize, outputImageSize, System.Drawing.Drawing2D.InterpolationMode.NearestNeighbor);
219 byte[] bytes = ImageUtilities.BitmapToArray(resized);
220 double[] inputImage = bytes.Select((pixel) => ((double)pixel / 255.0)).ToArray();
221
222 double[] result = neuralNetwork.Predict(inputImage);
223 int prediction = Utilities.ArgMax(result);
224
225 predictionTextBox.Text = $"My prediction is: {prediction}";
226
227 int iconSize = drawingHistoryDataGrid.Columns[0].Width;
228 Bitmap icon = ImageUtilities.Resize(bitmap, iconSize, iconSize);
229
230 int rowIdx = drawingHistoryDataGrid.Rows.Add();
231 var row = drawingHistoryDataGrid.Rows[rowIdx];
232 row.Cells[0].Value = icon;
233 row.Cells[1].Value = prediction;
234 row.Height = iconSize;
235
236 fullSizeBitmapHistory[row] = bitmap.Clone() as Bitmap;
237 }
238 }
