122 private void recognizeButton_Click(object sender, EventArgs e)
123 {
124 if (neuralNetwork == null)
125 {
126 predictionTextBox.Text = "Load or train your neural network first!";
127 }
128 else
129 {
130 Bitmap resized = ImageUtilities.Resize(bitmap, outputImageSide, outputImageSide, System.Drawing.Drawing2D.InterpolationMode.NearestNeighbor);
131 resized = ImageUtilities.Resize(bitmap, outputImageSide * 2, outputImageSide * 2, System.Drawing.Drawing2D.InterpolationMode.Bicubic);
132 resized = ImageUtilities.Resize(bitmap, outputImageSide, outputImageSide, System.Drawing.Drawing2D.InterpolationMode.NearestNeighbor);
133 byte[] bytes = ImageUtilities.BitmapToArray(resized);
134 double[] inputImage = bytes.Select((pixel) => ((double)pixel / 255.0)).ToArray();
135
136 double[] result = neuralNetwork.Predict(inputImage);
137 int prediction = Utilities.ArgMax(result);
138
139 predictionTextBox.Text = $"My prediction is: {prediction}";
140
141 int iconSize = drawingHistoryDataGrid.Columns[0].Width;
142 Bitmap icon = ImageUtilities.Resize(bitmap, iconSize, iconSize);
143
144 int rowIdx = drawingHistoryDataGrid.Rows.Add();
145 var row = drawingHistoryDataGrid.Rows[rowIdx];
146 row.Cells[0].Value = icon;
147 row.Cells[1].Value = prediction;
148 row.Height = iconSize;
149
150 fullSizeBitmapHistory[row] = bitmap.Clone() as Bitmap;
151 }
152 }
